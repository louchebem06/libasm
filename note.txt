%define MACH_SYSCALL(nb)	0x2000000 | nb
%define STDOUT				1
%define WRITE				4

section .data
hello:
	.string db "Hello World!", 10 // , 10 == \n
	.len equ $ - hello.string // taille de la string

// les code suivant seras dans la section text
section .text
	global start
	global _main
	extern _printf // existe tu la trouveras tkt

// start point d'entrer . ?? quivalant main ?
start:
	call _main
	ret

_main:
	push rbp // push sauvegarder // scope
	mov rbp, rsp // mov = rbp = rsp // registre de stack dans scope
	sub rsp, 16 // ajoute 16 octer a rsp
	lea rdi, [rel hello.string] // load hello world! in rdi
	call _printf // appel printf
	leave // moyen rapide de tout clear
	ret

_main:
	push rbp
	mov rbp, rsp
	sub rsp, 16
	move rdi, STDOUT
	lea rsi, [rel hello.string]
	mov rdx, hello.len
	move rax, MACH_SYSTEM(WRITE)
	syscall
	leave
	ret
